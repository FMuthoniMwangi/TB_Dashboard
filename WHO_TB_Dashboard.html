🎯 3. Outlier Treatment
▼
TB incidence varies enormously across countries. We use IQR-based Winsorisation per WHO region to cap extreme values while preserving valid epidemiological variation.
Python 
def winsorise_by_group(df, col, group_col='g_whoregion', factor=1.5):
    """Cap outliers within each WHO region using the IQR method (Winsorisation)."""
    result = df[col].copy()
    total_capped = 0
    for grp, grp_df in df.groupby(group_col):
        Q1  = grp_df[col].quantile(0.25)
        Q3  = grp_df[col].quantile(0.75)
        IQR = Q3 - Q1
        lo, hi = Q1 - factor * IQR, Q3 + factor * IQR
        mask    = grp_df.index
        n_cap   = ((result[mask] < lo) | (result[mask] > hi)).sum()
        total_capped += n_cap
        result[mask] = result[mask].clip(lower=lo, upper=hi)
    print(f'  {col:<22}: {total_capped} values winsorised')
    return result

print('Winsorisation summary (by WHO Region):')
for col in ['e_inc_100k', 'e_mort_100k', 'c_cdr']:
    df[col] = winsorise_by_group(df, col)

# ── Visualise before / after ─────────────────────────────────────────────────
fig, axes = plt.subplots(1, 3, figsize=(16, 5))
for ax, col in zip(axes, ['e_inc_100k', 'e_mort_100k', 'c_cdr']):
    ax.boxplot(df[col].dropna(), patch_artist=True,
               boxprops=dict(facecolor='#fc8d62', color='#333'))
    ax.set_title(f'{col}\nAfter Winsorisation', fontweight='bold')
plt.suptitle('After Outlier Treatment (Winsorised by WHO Region)',
             fontsize=14, fontweight='bold')

df.sort_values(['country', 'year'], inplace=True)

df['inc_lag1'] = df.groupby('country')['e_inc_100k'].shift(1)

df['inc_roll3'] = df.groupby('country')['e_inc_100k'].transform(

df['inc_pct_change'] = df.groupby('country')['e_inc_100k'].pct_change() * 100

🌲 6. Model Building — Random Forest
▼
Temporal train/test split (no data leakage): train on 2002–2021, test on 2022–2024. Fit a Random Forest Regressor with 500 trees. Safety checks remove any remaining inf/NaN before fitting.
Python 
FEATURES = [
    'year_index', 'region_enc', 'country_enc',
    'inc_lag1', 'inc_lag2', 'inc_roll3', 'inc_pct_change',
    'e_mort_100k', 'e_tbhiv_prct', 'c_newinc_100k', 'c_cdr', 'cfr_pct'
]
TARGET = 'e_inc_100k'

# ── Temporal split (respects time ordering — no data leakage) ────────────────
TRAIN_CUTOFF = 2021
train = df_model[df_model['year'] <= TRAIN_CUTOFF]
test  = df_model[df_model['year'] >  TRAIN_CUTOFF]

X_train = train[FEATURES].copy();  y_train = train[TARGET]
X_test  = test[FEATURES].copy();   y_test  = test[TARGET]

# ── Safety: replace any remaining inf / NaN with column medians ──────────────
for col in FEATURES:
    X_train[col] = X_train[col].replace([np.inf, -np.inf], np.nan)
    X_test[col]  = X_test[col].replace([np.inf, -np.inf], np.nan)
    med = X_train[col].median()
    X_train[col] = X_train[col].fillna(med)
    X_test[col]  = X_test[col].fillna(med)

print(f'Training samples : {len(X_train):,}  (years 2002–{TRAIN_CUTOFF})')
print(f'Testing samples  : {len(X_test):,}   (years {TRAIN_CUTOFF+1}–2024)')
print(f'Inf in X_train   : {np.isinf(X_train.values).sum()} | NaN: {np.isnan(X_train.values).sum()}')

# ── Random Forest Regressor ──────────────────────────────────────────────────
rf = RandomForestRegressor(
    n_estimators      = 500,     # number of trees
    max_depth         = 12,      # max tree depth
    min_samples_split = 4,       # min samples to split a node
    min_samples_leaf  = 2,       # min samples in leaf node
    max_features      = 'sqrt',  # features per split = √n_features
    random_state      = 42,
    n_jobs            = -1       # use all CPU cores
)
rf.fit(X_train, y_train)
print('\n✅ Random Forest trained (500 trees, max_depth=12)')

y_pred_train = rf.predict(X_train)
y_pred_test  = rf.predict(X_test)

def print_metrics(y_true, y_pred, label):
    mae  = mean_absolute_error(y_true, y_pred)
    rmse = np.sqrt(mean_squared_error(y_true, y_pred))
    r2   = r2_score(y_true, y_pred)
    mape = np.mean(np.abs((y_true - y_pred) / np.maximum(y_true, 1))) * 100
    print(f'  ── {label} ──')
    print(f'  R²   : {r2:.4f}')
    print(f'  MAE  : {mae:.2f} per 100k')
    print(f'  RMSE : {rmse:.2f} per 100k')
    print(f'  MAPE : {mape:.2f}%')
    return mae, rmse, r2, mape
